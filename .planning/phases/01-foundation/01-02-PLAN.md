2. Create `backend/app/core/cloudinary_client.py` with the configure function. The cloudinary SDK has a module-level `cloudinary.config(...)` call — wrap it so we configure once from `Settings`. Module body:
```python
"""Cloudinary SDK configuration.

The cloudinary Python SDK reads global state via cloudinary.config(...).
Call configure_cloudinary() once at FastAPI startup. The api_secret is
held only in module state and never exposed via any route response.
"""
from __future__ import annotations

import logging

import cloudinary

from app.config import Settings, get_settings

logger = logging.getLogger(__name__)

_configured: bool = False


def configure_cloudinary(settings: Settings | None = None) -> None:
    """Idempotent: configure the cloudinary SDK from Settings.

    Safe to call multiple times. Logs the cloud name on first config — never the secret.
    """
    global _configured
    if _configured:
        return

    s = settings or get_settings()
    if not (s.cloudinary_cloud_name and s.cloudinary_api_key and s.cloudinary_api_secret):
        logger.warning(
            "Cloudinary credentials are empty; configure_cloudinary() is a no-op until env vars are set."
        )
        return

    cloudinary.config(
        cloud_name=s.cloudinary_cloud_name,
        api_key=s.cloudinary_api_key,
        api_secret=s.cloudinary_api_secret,
        secure=True,
    )
    _configured = True
    logger.info("Cloudinary configured (cloud=%s)", s.cloudinary_cloud_name)


def is_configured() -> bool:
    return _configured
```

```python
"""Cloudinary signed-upload signature endpoint.

Phase 1 (this plan): real implementation. Authentication is optional in Phase 1
(get_current_user uses auto_error=False) so the endpoint can be smoke-tested
before Phase 2 wires real JWT validation. Phase 2 must flip auto_error=True.
"""
from __future__ import annotations

import logging
import time

import cloudinary.utils
from fastapi import APIRouter, Depends, HTTPException, status

from app.config import Settings, get_settings
from app.core.cloudinary_client import configure_cloudinary
from app.dependencies import get_current_user  # auto_error=False per plan 01-01

logger = logging.getLogger(__name__)

router = APIRouter()

_FOLDER = "revvbase/listings"


@router.post("/signature")
async def get_upload_signature(
    settings: Settings = Depends(get_settings),
    # Phase 1: auth is optional (get_current_user is a stub). Phase 2 will tighten this.
    _user=Depends(lambda: None),  # placeholder dependency; real get_current_user wired in Phase 2
):
    """Generate a fresh Cloudinary signed-upload payload.

    Returns: { signature, timestamp, api_key, cloud_name, folder }.
    The api_secret is consumed by api_sign_request and is never returned or logged.
    """
    # Reject early when secrets aren't set — never let an empty-secret signature reach the client.
    if not (settings.cloudinary_cloud_name and settings.cloudinary_api_key and settings.cloudinary_api_secret):
        raise HTTPException(
            status_code=status.HTTP_503_SERVICE_UNAVAILABLE,
            detail="Cloudinary not configured",
        )

    # Ensure SDK is configured (idempotent — safe even if main.py already called it).
    configure_cloudinary(settings)

    timestamp = int(time.time())  # fresh epoch seconds — Cloudinary requires <1h old
    params_to_sign = {
        "timestamp": timestamp,
        "folder": _FOLDER,
    }

    signature = cloudinary.utils.api_sign_request(
        params_to_sign,
        settings.cloudinary_api_secret,
    )

    return {
        "signature": signature,
        "timestamp": timestamp,
        "api_key": settings.cloudinary_api_key,
        "cloud_name": settings.cloudinary_cloud_name,
        "folder": _FOLDER,
    }
```

```python
  Send:    POST https://control.msg91.com/api/v5/otp
           Headers: { "authkey": "<MSG91_AUTH_KEY>", "Content-Type": "application/json" }
           Query:   template_id=<MSG91_TEMPLATE_ID>&mobile=<phone-with-country-code>
           Success: { "type": "success", "request_id": "..." }
  Verify:  POST https://control.msg91.com/api/v5/otp/verify
           Headers: { "authkey": "<MSG91_AUTH_KEY>" }
           Query:   otp=<6-digit>&mobile=<phone-with-country-code>
           Success: { "type": "success", "message": "OTP verified successfully" }
"""
from __future__ import annotations

import logging
from typing import Any

import httpx
from fastapi import HTTPException, status

from app.config import Settings

logger = logging.getLogger(__name__)

_MSG91_BASE = "https://control.msg91.com/api/v5/otp"
_MOCK_OTP_CODE = "123456"


async def send_otp(phone: str, settings: Settings) -> dict[str, Any]:
    """Send an OTP to `phone`. In mock mode, returns success without network calls.

    Returns: { "success": True, "mock": <bool>, ...optional MSG91 request_id }.
    Raises: HTTPException on MSG91 transport / contract failure.
    """
    if settings.mock_otp:
        logger.info("MOCK OTP: use %s for %s", _MOCK_OTP_CODE, phone)
        return {"success": True, "mock": True}

    # Real MSG91 path
    if not (settings.msg91_auth_key and settings.msg91_template_id):
        raise HTTPException(
            status_code=status.HTTP_503_SERVICE_UNAVAILABLE,
            detail="MSG91 not configured",
        )

    try:
        async with httpx.AsyncClient(timeout=10.0) as client:
            resp = await client.post(
                _MSG91_BASE,
                headers={
                    "authkey": settings.msg91_auth_key,
                    "Content-Type": "application/json",
                },
                params={
                    "template_id": settings.msg91_template_id,
                    "mobile": phone,
                },
            )
    except httpx.HTTPError as exc:
        # Do NOT echo the upstream error body — it may contain authkey-shaped data.
        logger.exception("MSG91 send-otp transport error")
        raise HTTPException(
            status_code=status.HTTP_502_BAD_GATEWAY,
            detail="OTP provider unavailable",
        ) from exc

    if resp.status_code != 200:
        logger.warning("MSG91 send-otp non-200: %s", resp.status_code)
        raise HTTPException(
            status_code=status.HTTP_502_BAD_GATEWAY,
            detail="OTP provider returned an error",
        )

    body = resp.json() if resp.content else {}
    if body.get("type") != "success":
        raise HTTPException(
            status_code=status.HTTP_502_BAD_GATEWAY,
            detail="OTP provider rejected the request",
        )

    return {"success": True, "mock": False, "request_id": body.get("request_id")}


async def verify_otp(phone: str, otp: str, settings: Settings) -> bool:
    """Verify `otp` for `phone`. Returns True on valid, False on invalid.

    Raises HTTPException only for transport/contract failures in real mode.
    """
    if settings.mock_otp:
        return otp == _MOCK_OTP_CODE

    if not settings.msg91_auth_key:
        raise HTTPException(
            status_code=status.HTTP_503_SERVICE_UNAVAILABLE,
            detail="MSG91 not configured",
        )

    try:
        async with httpx.AsyncClient(timeout=10.0) as client:
            resp = await client.post(
                f"{_MSG91_BASE}/verify",
                headers={"authkey": settings.msg91_auth_key},
                params={"otp": otp, "mobile": phone},
            )
    except httpx.HTTPError as exc:
        logger.exception("MSG91 verify-otp transport error")
        raise HTTPException(
            status_code=status.HTTP_502_BAD_GATEWAY,
            detail="OTP provider unavailable",
        ) from exc

    if resp.status_code != 200:
        # MSG91 returns 200 even for invalid OTPs and signals via body.type — non-200 is a real error.
        return False

    body = resp.json() if resp.content else {}
    return body.get("type") == "success"
```
